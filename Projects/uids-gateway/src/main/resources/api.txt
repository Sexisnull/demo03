# QQ-QQAuth接口（获取code）需要替换 CLIENT_ID和REDIRECT_URL及STATE
qq_auth_url=https://graph.qq.com/oauth2.0/authorize?response_type=CODE&client_id=CLIENT_ID&redirect_uri=REDIRECT_URI&state=1

# QQ-QQAuth接口（获取access_tokenUrl）需要替换GRANT_TYPE、CLIENT_ID、CLIENT_SECRET、CODE、REDIRECT_URL
get_qq_accesstoken_url=https://graph.qq.com/oauth2.0/token?grant_type=GRANT_TYPE&client_id=CLIENT_ID&client_secret=CLIENT_SECRET&code=CODE&redirect_uri=REDIRECT_URL

# QQ-QQAuth接口（获取openid）需要替换ACCESS_TOKEN
get_qq_openid_url=https://graph.qq.com/oauth2.0/me?access_token=ACCESS_TOKEN

# 绑定QQopenid与系统UserId
